\begin{frame}[shrink=7]{Two-workflow design}
  \begin{columns}[T]
    \column{0.48\textwidth}
      \textbf{Workflow 1 --- \texttt{fair-validate.yml}}\\
      \textit{Trigger: push / pull request (paths: dmp.json)}
      \begin{itemize}\setlength\itemsep{2pt}
        \item Structural validation via \texttt{madmpy}
        \item Runs each policy rule against every dataset
        \item Exit 1 if any error-level rule fails
        \item Writes rule-by-rule table to GHA step summary
      \end{itemize}
      \vspace{0.15cm}
      \begin{alertblock}{Governance requirement}
        \footnotesize Branch protection must require this check ---
        otherwise a PI can create a release without validation passing
      \end{alertblock}

    \column{0.48\textwidth}
      \textbf{Workflow 2 --- \texttt{fair-publish.yml}}\\
      \textit{Trigger: GitHub release published}
      \begin{itemize}\setlength\itemsep{2pt}
        \item Trusts the validation gate (no re-validation)
        \item New dataset $\to$ create new Zenodo record
        \item Known dataset $\to$ new version (via \texttt{zenodo\_state.json})
        \item Commits updated state file back to repo
      \end{itemize}
      \vspace{0.1cm}
      \begin{block}{Why split?}
        \footnotesize Validation gives fast feedback on every push.
        Publishing only happens at a deliberate version event.
        The gate cannot be bypassed.
      \end{block}
  \end{columns}
\end{frame}

\section{Rule system}
%==============================================================================

\begin{frame}[fragile,shrink=6]{Built-in rules and policy}
  \begin{columns}[T]
    \column{0.48\textwidth}
      \textbf{Six built-in FAIR rules:}
      \begin{itemize}\setlength\itemsep{3pt}
        \item Creator ORCID present \hfill\textit{F1, A1}
        \item License present \& recognized \hfill\textit{R1.1}
        \item At least one keyword \hfill\textit{F2}
        \item Description $\geq$ N chars \hfill\textit{F2, R1}
        \item Access rights stated \hfill\textit{A1}
        \item Dataset identifier present \hfill\textit{F1}
      \end{itemize}
      \vspace{0.25cm}
      \textbf{Open:} \texttt{github.com/in-silico-institute/}\\
      \texttt{.github/blob/main/.github/policies/}\\
      \texttt{fair-policy.yml}

    \column{0.50\textwidth}
      \begin{lstlisting}[style=yaml, caption={fair-policy.yml (in-silico-institute)}]
rules:
  - name: Creator ORCID present
    severity: warning
  - name: License present and recognized
    severity: error
  - name: At least one keyword
    severity: warning
  - name: Description length
    severity: warning
    config:
      min_chars: 100
  - name: Access rights stated
    severity: error
  - name: Dataset identifier present
    severity: error

targets:
  - zenodo
      \end{lstlisting}
  \end{columns}
\end{frame}

\begin{frame}[fragile]{Extending with domain rules}
  Institutions add domain-specific rules without modifying the core package.

  \vspace{0.05cm}
  \begin{columns}[T]
    \column{0.50\textwidth}
      \begin{lstlisting}[style=python, caption={Example: SNOMED CT rule}]
# myorg_rules/biomedical.py
from fair_publish.rules import RuleResult

class SnomedCtKeyword:
    name = "SNOMED CT keyword present"
    severity = "error"
    fair = "I1, R1.3"

    def __call__(self, dmp, dataset):
        kws = getattr(dataset, "keyword", []) or []
        ok = any("snomed" in k.lower()
                 for k in kws)
        return RuleResult(
            passed=ok, rule=self.name,
            severity=self.severity,
            field="dataset.keyword",
            fair=self.fair,
            message="" if ok else
              "No SNOMED CT keyword found.",
        )
      \end{lstlisting}

    \column{0.47\textwidth}
      \begin{lstlisting}[style=yaml, caption={Register in policy.yml}]
rules:
  - name: License present and recognized
    severity: error
  - name: At least one keyword
    severity: error
  - name: Description length
    severity: warning
    config:
      min_chars: 150

extra_rules:
  - module: myorg_rules.biomedical
    class: SnomedCtKeyword
    severity: error

targets:
  - zenodo
      \end{lstlisting}
      \vspace{0.05cm}
      \footnotesize One policy YAML per institution.\\
      Propagates to all project repos automatically.
  \end{columns}
\end{frame}

\section{Summary}

\begin{frame}[shrink=10]{Summary and limitations}
  \begin{columns}[T]
    \column{0.52\textwidth}
      \textbf{What the pipeline provides:}
      \begin{itemize}\setlength\itemsep{1pt}
        \item Pluggable rule protocol: built-in FAIR rules
              + institution domain rules
        \item One YAML policy at org level, inherited by all projects
        \item Per-dataset lifecycle tracking (\texttt{zenodo\_state.json}):\\
              new record vs.\ new version decided automatically
        \item Two-workflow design: validate on push,
              deploy on release
        \item CLI for local validation and sandbox simulation
        \item Low entry barrier: two files per project repo
      \end{itemize}

    \column{0.45\textwidth}
      \begin{alertblock}{Current limitations}
        \begin{itemize}\setlength\itemsep{2pt}
          \item DMP authoring tool is external (assumed to exist)
          \item Repos must be institution-owned
          \item Branch protection must be configured separately
          \item Metadata only --- no data file upload yet
          \item Policy evolution not automated
          \item No error notification mechanism
        \end{itemize}
      \end{alertblock}
      \vspace{0.1cm}
      \begin{block}{Live demo org}
        \footnotesize
        \url{https://github.com/in-silico-institute}\\[4pt]
        Package: \texttt{in-silico-institute/fair-publish}
      \end{block}
  \end{columns}
